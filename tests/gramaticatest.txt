6. TABELA LL(1)

[comando]
  com EPAR: comando -> EPAR conteudo DPAR

[cont_comando]
  com EPAR: cont_comando -> comando operador_final
  com MEM: cont_comando -> MEM fim_mem
  com NUM: cont_comando -> NUM operador_final

[cont_mem]
  com DPAR: cont_mem -> ε
  com EPAR: cont_mem -> comando operador_final
  com MEM: cont_mem -> MEM fim_mem
  com NUM: cont_mem -> NUM operador_final

[cont_num]
  com EPAR: cont_num -> comando operador_final
  com MEM: cont_num -> MEM fim_mem
  com NUM: cont_num -> NUM operador_final
  com RES: cont_num -> RES

[conteudo]
  com EPAR: conteudo -> comando cont_comando
  com MEM: conteudo -> MEM cont_mem
  com NUM: conteudo -> NUM cont_num

[fim_mem]
  com DPAR: fim_mem -> ε
  com ENQUANTO: fim_mem -> operador_final
  com OP: fim_mem -> operador_final
  com OPREL: fim_mem -> operador_final
  com SE: fim_mem -> operador_final

[lista_comandos]
  com END: lista_comandos -> ε
  com EPAR: lista_comandos -> comando lista_comandos

[operador_final]
  com ENQUANTO: operador_final -> ENQUANTO
  com OP: operador_final -> OP
  com OPREL: operador_final -> OPREL
  com SE: operador_final -> SE

[programa]
  com START: programa -> START lista_comandos END
